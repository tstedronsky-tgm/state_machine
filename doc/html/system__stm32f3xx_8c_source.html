244 void SystemCoreClockUpdate (void)
245 {
246  uint32_t tmp = 0, pllmull = 0, pllsource = 0, predivfactor = 0;
247 
248  /* Get SYSCLK source -------------------------------------------------------*/
249  tmp = RCC->CFGR & RCC_CFGR_SWS;
250 
251  switch (tmp)
252  {
253  case RCC_CFGR_SWS_HSI: /* HSI used as system clock */
254  SystemCoreClock = HSI_VALUE;
255  break;
256  case RCC_CFGR_SWS_HSE: /* HSE used as system clock */
257  SystemCoreClock = HSE_VALUE;
258  break;
259  case RCC_CFGR_SWS_PLL: /* PLL used as system clock */
260  /* Get PLL clock source and multiplication factor ----------------------*/
261  pllmull = RCC->CFGR & RCC_CFGR_PLLMUL;
262  pllsource = RCC->CFGR & RCC_CFGR_PLLSRC;
263  pllmull = ( pllmull >> 18) + 2;
264 
265  if (pllsource == RCC_CFGR_PLLSRC_HSI_DIV2)
266  {
267  /* HSI oscillator clock divided by 2 selected as PLL clock entry */
268  SystemCoreClock = (HSI_VALUE >> 1) * pllmull;
269  }
270  else
271  {
272  predivfactor = (RCC->CFGR2 & RCC_CFGR2_PREDIV) + 1;
273  /* HSE oscillator clock selected as PREDIV1 clock entry */
274  SystemCoreClock = (HSE_VALUE / predivfactor) * pllmull;
275  }
276  break;
277  default: /* HSI used as system clock */
278  SystemCoreClock = HSI_VALUE;
279  break;
280  }
281  /* Compute HCLK clock frequency ----------------*/
282  /* Get HCLK prescaler */
283  tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
284  /* HCLK clock frequency */
285  SystemCoreClock >>= tmp;
286 }
